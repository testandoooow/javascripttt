numero7.addEventListener("click", (event) => {

    limparTexto();
    paragrafo.innerText = paragrafo.innerText+""+numero;

numero2.addEventListener("click", (event) => {

    limparTexto();
    paragrafo.innerText = paragrafo.innerText+""+numero;
    verificar();
  });
  
  adicao.addEventListener("click", (event) => {
    adicaoB = true;
    paragrafo.innerText = "+";
  });

  subtracao.addEventListener("click", (event) => {
    subtracaoB = true;
    paragrafo.innerText = "-";
  });

  multiplicacao.addEventListener("click", (event) => {
    multiplicacaoB = true;
    paragrafo.innerText = "*";
  });

  divisao.addEventListener("click", (event) => {
    divisaoB = true;
    paragrafo.innerText = "รท";
  });

  resultado.addEventListener("click", (event) => {
    console.log(outroNumero);
  });

  limpar.addEventListener("click", (event) => {
    adicaoB = false;
    subtracaoB = false;
    multiplicacaoB = false;
    divisaoB = false;
    paragrafo.innerText = "";
    outroNumero = 0;
  });

  function verificar(){
      if(outroNumero === 0){
          outroNumero = numero;
        }
        if(adicaoB){
            outroNumero += Number(paragrafo.innerText);
        }
        if(subtracaoB){
            outroNumero -= Number(paragrafo.innerText);
        }
        if(multiplicacaoB){
            outroNumero *= Number(paragrafo.innerText);
        }
        if(divisaoB){
            outroNumero /= Number(paragrafo.innerText);
        }
    }

  function limparTexto(){
    if(paragrafo.innerText == "+" || paragrafo.innerText == "-" || paragrafo.innerText == "*" || paragrafo.innerText == "รท"){
        paragrafo.innerText = "";
    }
  }
